        firebase.initializeApp(firebaseConfig);
        const auth = firebase.auth();
        const db = firebase.firestore();

        // State
        let currentArea = 'frozen';  // 'frozen' atau 'chiller'
        let appData = {
            frozen: { sections: {}, baskets: {}, transactions: [] },
            chiller: { sections: {}, baskets: {}, transactions: [] }
        };
        let currentTab = 'dashboard';
        let unsubscribe = null;

        // Default sections
        const defaultFrozenSections = {
            'RTL': [1,2,17,21,23,31,32,33,36],
            'IMP': [6,7,8,10,18,19,30,38,46,55,62,63,66,68,69,70,71,73],

        const defaultChillerSections = {

        function getDefaultSections(area) {
            return area === 'frozen' ? JSON.parse(JSON.stringify(defaultFrozenSections)) :
                   JSON.parse(JSON.stringify(defaultChillerSections));
        }

        // Auth
        function login() {
            const email = document.getElementById('loginEmail').value.trim();
            const pass = document.getElementById('loginPassword').value.trim();
            auth.signInWithEmailAndPassword(email, pass)
                .catch(e => document.getElementById('loginError').textContent = 'Gagal: ' + e.message);
        }
        function logout() { auth.signOut(); }

        auth.onAuthStateChanged(user => {
            if (user) {
                document.getElementById('loginScreen').style.display = 'none';
                document.getElementById('appScreen').style.display = 'block';
                document.getElementById('currentUserName').textContent = user.email;
                listenToFirestore();
            } else {
                document.getElementById('loginScreen').style.display = 'block';
                document.getElementById('appScreen').style.display = 'none';
                if (unsubscribe) unsubscribe();
            }
        });

        // Firestore
        function listenToFirestore() {
            unsubscribe = db.collection('shared').doc('data').onSnapshot(doc => {
                if (doc.exists) {
                    const data = doc.data();
                    appData.frozen = data.frozen || { sections: getDefaultSections('frozen'), baskets: {}, transactions: [] };
                    appData.chiller = data.chiller || { sections: getDefaultSections('chiller'), baskets: {}, transactions: [] };
                    syncBasketsWithSections('frozen');
                    syncBasketsWithSections('chiller');
                    if (!data.frozen) saveToFirestore(); // simpan default pertama kali
                    renderCurrentTab();
                } else {
                    // inisialisasi
                    appData.frozen = { sections: getDefaultSections('frozen'), baskets: {}, transactions: [] };
                    appData.chiller = { sections: getDefaultSections('chiller'), baskets: {}, transactions: [] };
                    syncBasketsWithSections('frozen');
                    syncBasketsWithSections('chiller');
                    saveToFirestore();
                }
            });
        }

        function saveToFirestore() {
            db.collection('shared').doc('data').set({
                frozen: appData.frozen,
                chiller: appData.chiller
            }, { merge: true });
        }

        function syncBasketsWithSections(area) {
            const sections = appData[area].sections;
            const baskets = appData[area].baskets;
            for (const [sec, nums] of Object.entries(sections)) {
                nums.forEach(no => {
                    if (!baskets[no]) baskets[no] = { section: sec, items: [] };
                    else baskets[no].section = sec;
                });
            }
        }

        function currentSections() { return appData[currentArea].sections; }
        function currentBaskets() { return appData[currentArea].baskets; }
        function currentTransactions() { return appData[currentArea].transactions; }

        function switchArea() {
            currentArea = document.getElementById('areaSelect').value;
            document.getElementById('areaTitle').textContent = currentArea === 'frozen' ? 'Frozen' : 'Chiller';
            renderCurrentTab();
        }

        // Helpers
        function stackCount(no) { return (currentBaskets()[no]?.items || []).length; }
        function basketStatus(no) {
            const c = stackCount(no);
            if (c >= 5) return 'red';
            if (c >= 3) return 'yellow';
            return 'green';
        }
        function showToast(msg) {
            const t = document.getElementById('toast');
            t.textContent = msg;
            t.classList.add('show');
            setTimeout(() => t.classList.remove('show'), 2500);
        }
        function getAvailableBasket(section) {
            const nums = currentSections()[section] || [];
            return nums.find(n => stackCount(n) < 5) || null;
        }
        function logTransaction(type, detail) {
            currentTransactions().unshift({
                id: Date.now(),
                timestamp: new Date().toISOString(),
                type,
                detail,
                user: auth.currentUser?.email || 'unknown'
            });
            if (currentTransactions().length > 200) currentTransactions().pop();
            saveToFirestore();
        }

        // Render
        const container = document.getElementById('appContainer');
        function renderCurrentTab() {
            switch(currentTab) {
                case 'dashboard': renderDashboard(); break;
                case 'in': renderInputForm(); break;
                case 'out': renderOut(); break;
                case 'history': renderHistory(); break;
                case 'manage': renderManageBaskets(); break;
            }
        }

        function renderDashboard() {
            let html = '';
            for (const [section, keranjang] of Object.entries(currentSections())) {
                const cards = keranjang.sort((a,b)=>a-b).map(no => {
                    const status = basketStatus(no);
                    const count = stackCount(no);
                    let bars = '';
                    for (let i=0; i<5; i++) {
                        let cls = 'stack-bar';
                        if (i < count) {
                            cls += ' filled';
                            if (count >=5) cls += ' danger';
                            else if (count >=3) cls += ' warning';
                        }
                        bars += `<div class="${cls}"></div>`;
                    }
                    return `<div class="basket-card ${status}" onclick="showBasketDetail(${no})" title="Keranjang ${no} - ${count}/5">
                        <span class="basket-number">${no}</span>
                        <div class="stack-visual">${bars}</div>
                        <span class="stack-label">${count}/5</span>
                    </div>`;
                }).join('');
                html += `<div class="card"><div class="section-header"><h3>📍 ${section}</h3><small>${keranjang.length} keranjang</small></div><div class="basket-grid">${cards}</div></div>`;
            }
            container.innerHTML = html;
        }

        function showBasketDetail(no) {
            const basket = currentBaskets()[no];
            const count = stackCount(no);
            const status = basketStatus(no);
            let itemsHtml = basket && basket.items.length ? basket.items.map((item, idx) => `
                <div class="item-row">
                    <div><strong>${item.nama}</strong> (${item.qty})<br><small>${item.jenis || ''}</small></div>
                    <button class="btn btn-outline" onclick="keluarkanItem(${no}, ${idx})">Keluar</button>
                </div>`).join('') : '<div class="empty-state">Keranjang kosong</div>';
            container.innerHTML = `
                <div class="card">
                    <div style="display:flex; justify-content:space-between;">
                        <h3>Keranjang ${no}</h3>
                        <button class="btn btn-outline" onclick="renderDashboard()">← Kembali</button>
                    </div>
                    <p>Status: ${status==='red'?'🔴 Penuh':status==='yellow'?'🟡 Hampir Penuh':'🟢 Tersedia'} | Tumpukan: ${count}/5</p>
                    ${itemsHtml}
                </div>`;
        }

        function keluarkanItem(no, idx) {
            const basket = currentBaskets()[no];
            if (!basket) return;
            const item = basket.items[idx];
            if (confirm(`Keluarkan "${item.nama}" dari keranjang ${no}?`)) {
                basket.items.splice(idx, 1);
                logTransaction('out', { nama: item.nama, qty: item.qty, keranjang: no, section: basket.section });
                saveToFirestore();
                showToast('✅ Barang dikeluarkan');
                showBasketDetail(no);
            }
        }

        function renderInputForm() {
            const sectionOpt = Object.keys(currentSections()).map(s => `<option value="${s}">${s}</option>`).join('');
            container.innerHTML = `
                <div class="card">
                    <h3>📥 Barang Masuk</h3>
                    <form id="inForm" onsubmit="submitIn(event)">
                        <div class="form-group"><label>Nama Barang</label><input type="text" id="inNama" list="namaList" required><datalist id="namaList"></datalist></div>
                        <div class="form-group"><label>Jumlah</label><input type="text" id="inQty" placeholder="3 Pack" required></div>
                        <div class="form-group"><label>Jenis</label><select id="inJenis"><option value="IIP">IIP</option><option value="RM">RM</option><option value="TH">TH</option></select></div>
                        <div class="form-group"><label>Section</label><select id="inSection" onchange="updateKeranjangSelect()">${sectionOpt}</select></div>
                        <div class="form-group"><label>No Keranjang</label><select id="inKeranjang"><option value="">-- Otomatis --</option></select></div>
                        <button type="submit" class="btn btn-primary" style="width:100%">Simpan</button>
                    </form>
                </div>`;
            updateKeranjangSelect();
            updateNamaDatalist();
        }

        function updateKeranjangSelect() {
            const section = document.getElementById('inSection')?.value;
            const select = document.getElementById('inKeranjang');
            if(!select) return;
            select.innerHTML = '<option value="">-- Otomatis --</option>';
            if(section) {
                (currentSections()[section]||[]).sort((a,b)=>a-b).forEach(no => {
                    const c = stackCount(no);
                    select.innerHTML += `<option value="${no}" ${c>=5?'disabled':''}>${no} (${c}/5)${c>=5?' PENUH':''}</option>`;
                });
            }
        }

        function updateNamaDatalist() {
            const set = new Set();
            currentTransactions().forEach(t => { if(t.detail?.nama) set.add(t.detail.nama); });
            document.getElementById('namaList').innerHTML = [...set].map(n => `<option value="${n}">`).join('');
        }

        function submitIn(e) {
            e.preventDefault();
            const nama = document.getElementById('inNama').value.trim();
            const qty = document.getElementById('inQty').value.trim();
            const jenis = document.getElementById('inJenis').value;
            const section = document.getElementById('inSection').value;
            let keranjang = document.getElementById('inKeranjang').value;
            if(!nama||!qty) return showToast('Lengkapi data!');
            if(!keranjang) {
                const auto = getAvailableBasket(section);
                if(!auto) return showToast(`Semua keranjang di ${section} penuh!`);
                keranjang = auto;
            }
            const no = parseInt(keranjang);
            if(stackCount(no)>=5) return showToast('Keranjang penuh!');
            if(!currentBaskets()[no]) currentBaskets()[no] = { section, items:[] };
            currentBaskets()[no].items.push({ nama, qty, jenis, tglMasuk: new Date().toISOString() });
            logTransaction('in', { nama, qty, jenis, keranjang:no, section });
            saveToFirestore();
            showToast('✅ Barang masuk');
            renderInputForm();
        }

        function renderOut() {
            let items = [];
            Object.entries(currentBaskets()).forEach(([no, basket]) => {
                basket.items.forEach((item, idx) => items.push({ no:parseInt(no), section:basket.section, item, idx }));
            });
            if(!items.length) {
                container.innerHTML = '<div class="card"><h3>📤 Barang Keluar</h3><div class="empty-state">Tidak ada barang</div></div>';
                return;
            }
            container.innerHTML = `<div class="card"><h3>📤 Barang Keluar</h3>` +
                items.map(e => `
                    <div class="item-row">
                        <div><strong>${e.item.nama}</strong> (${e.item.qty})<br><small>Keranjang ${e.no} [${e.section}]</small></div>
                        <button class="btn btn-danger" onclick="keluarkanItem(${e.no},${e.idx});renderOut();">Keluar</button>
                    </div>`).join('') + '</div>';
        }

        function renderHistory() {
            const list = currentTransactions().slice(0,50).map(t => {
                const icon = t.type==='in'?'📥':'📤';
                return `<div class="item-row"><div>${icon} <strong>${t.detail.nama}</strong> (${t.detail.qty})<br><small>${new Date(t.timestamp).toLocaleString('id-ID')} | ${t.user}</small></div></div>`;
            }).join('');
            container.innerHTML = `<div class="card"><h3>📋 Riwayat</h3>${list||'<div class="empty-state">Belum ada aktivitas</div>'}</div>`;
        }

        function renderManageBaskets() {
            let html = '<div class="card"><h3>⚙️ Atur Nomor Keranjang</h3>';
            for(const [section, nums] of Object.entries(currentSections())) {
                const chips = nums.sort((a,b)=>a-b).map(no => {
                    const count = stackCount(no);
                    const canDel = count===0;
                    return `<span class="chip">${no} <span style="cursor:pointer" onclick="${canDel?`removeBasket('${section}',${no})`:'showToast(\'Keranjang berisi\')'}">${canDel?'🗑️':'🔒'}</span></span>`;
                }).join('');
                html += `<div style="margin-bottom:16px"><strong>📍 ${section}</strong><div class="chip-list">${chips}</div>
                    <input type="number" id="new_${section}" placeholder="Nomor baru" style="width:100px;">
                    <button class="btn btn-primary btn-sm" onclick="addBasket('${section}')">+ Tambah</button></div>`;
            }
            html += '</div>';
            container.innerHTML = html;
        }

        function addBasket(section) {
            const input = document.getElementById(`new_${section}`);
            if(!input) return;
            const no = parseInt(input.value);
            if(isNaN(no)||no<=0) return showToast('Nomor tidak valid');
            for(const [sec, nums] of Object.entries(currentSections())) {
                if(nums.includes(no)) return showToast(`Nomor ${no} sudah ada di ${sec}`);
            }
            currentSections()[section].push(no);
            if(!currentBaskets()[no]) currentBaskets()[no] = { section, items:[] };
            else currentBaskets()[no].section = section;
            saveToFirestore();
            input.value = '';
            renderManageBaskets();
            showToast(`✅ ${no} ditambahkan ke ${section}`);
        }

        function removeBasket(section, no) {
            if(stackCount(no)>0) return showToast('Keranjang masih berisi');
            if(!confirm(`Hapus keranjang ${no}?`)) return;
            const idx = currentSections()[section].indexOf(no);
            if(idx>-1) currentSections()[section].splice(idx,1);
            if(currentBaskets()[no]?.items.length===0) delete currentBaskets()[no];
            saveToFirestore();
            renderManageBaskets();
            showToast('✅ Dihapus');
        }

        // Tab navigation
        document.querySelectorAll('.tab-btn').forEach(btn => {
            btn.addEventListener('click', function() {
                document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                currentTab = this.dataset.tab;
                renderCurrentTab();
            });
        });

        // Initial
        document.getElementById('areaSelect').value = currentArea;
    </script>
